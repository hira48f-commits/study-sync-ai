BULLET NOTES BY CATEGORY`); lines.push(`---------------------------`); note.materials.bulletNotes.forEach((cat) => { lines.push(`Category: ${cat.category}`); cat.items.forEach((item) => lines.push(` • ${item}`)); lines.push(``); }); lines.push(`4.
